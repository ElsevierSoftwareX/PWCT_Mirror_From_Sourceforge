		Define Procedure Mypers
			Code Block
				num1 = CONVERT win1.text1.value TO Numeric
				num2 = CONVERT win1.text2.value TO Numeric
				pers = (num1/num2)*100
				pers = CONVERT pers TO STRING
				pers = All Trim pers
				win1.text3.Value := pers+" "+"%"
			End of Procedure

	procedureclear
		Define Procedure Clear
			Code Block
				num1 = CONVERT win1.text1.value TO Numeric
				num2 = CONVERT win1.text2.value TO Numeric
				num3 = CONVERT win1.text3.value TO Numeric
				num1 = 0
				num2 = 0
				num3 = 0
				num1 = CONVERT num1 TO STRING
				num2 = CONVERT num2 TO STRING
				num3 = CONVERT num3 TO STRING
				num1 = All Trim num1
				num2 = All Trim num2
				num3 = All Trim num3
				win1.text1.Value := num1
				win1.text2.Value := num2
				win1.text3.Value := num3
			End of Procedure
