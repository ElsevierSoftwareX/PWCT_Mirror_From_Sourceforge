	Define Procedure search
		Code Block
			cName = win1.textName.Value
			oCon = Create New ADODB Connection
			Open ADODB Connection oCon , Connection String : 'Provider=MSDASQL.1;Persist Security Info=False;Extended Properties="dsn=MS Access Database;uid=";Initial Catalog=data.mdb'
			oRS = Create New ADODB RecordSet Object
			Open Recordset oRS , Connection oCon , CursorType 1 , SQL Statement "Select * from tel where name like '" + cName + "%'"
			mymax := ADODB RecordSet (oRS) , Get Records Count
			win1.list1.DeleteAllItems ( )
			IF ( mymax > 0 )
				Code Block
					FOR LOOP ( From x = 1 To mymax Step 1)

							vname = ADODB RecordSet (oRS) Get Field ("name") Value
							vaddress = ADODB RecordSet (oRS) Get Field ("address") Value

					Code Block

				END OF IF STATEMENT
		End of Procedure
