	Define Window  ( win1 )

		Window Properties
		Window Controls
			Define TextBox  ( textadd  )
				TextBox EVENTS
				TextBox Properties
			Define TextBox  ( textcount  )
				TextBox EVENTS
				TextBox Properties
			Define New Button ( btnadd )
				Button Events
					Event: ON CLICK
				Button Properties
			Define New Button ( btncount )
				Button Events
					Event: ON CLICK
				Button Properties
			Define New ListBox ( list1 )
				ListBox Event
				ListBox Properties
		End Of Window
		Activate window
	Procedures
		Define Procedure myadd
			Code Block
				mytext = win1.textadd.Value
				win1.list1.AddItem ( mytext )
			End of Procedure
		Define Procedure mycount
			Code Block
				result = 0
				TheWord = win1.textcount.Value
				wordscount = win1.list1.ItemCount
				IF ( wordscount=0 )
					Code Block
						Show Message "Please, Enter any word" Title "Sorry" Type: Message Box
						Return 0
					END OF IF STATEMENT
				mycount = win1.list1.item(1)
				FOR LOOP ( From x = 1 To wordscount Step 1)
					Code Block
						myword = win1.list1.item(x)
						IF ( myword=TheWord )
							Code Block
								result = result+1
							END OF IF STATEMENT
					End of For Loop
				result = CONVERT result TO STRING
				Show Message mycount Title "The number of repeated word" Type: Message Box
			End of Procedure
