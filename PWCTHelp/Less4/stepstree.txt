Procedures
	Define Procedure connect
		Code Block
			Public oCon,oRS
			oCon = Create New ADODB Connection
			Open ADODB Connection oCon , Connection String : Initial Catalog=data.mdb'
			oRS = Create New ADODB RecordSet Object
			Open Recordset oRS , Connection oCon , CursorType 1 , SQL Statement "Select * from tel"
		End of Procedure
	Define Procedure disconnect
		Code Block
			Close ADODB RecordSet oRS
			Close ADODB Connection oCon
		End of Procedure
	Define Procedure closesystem
		Code Block
			Call Procedure disconnect
			win1.Release ( )
		End of Procedure
	Define Procedure showrecord
		Code Block
			mymax := ADODB RecordSet (oRS) , Get Records Count
			IF ( mymax > 0 )
				Code Block
					cname = ADODB RecordSet (oRS) Get Field ("name") Value
					caddress = ADODB RecordSet (oRS) Get Field ("address") Value
					cphone = ADODB RecordSet (oRS) Get Field ("phone") Value
				ELSE
					Code Block
						cname = ""
						caddress = ""
						cphone = ""
				END OF IF STATEMENT
			win1.textName.Value := cname
			win1.textAddress.Value := cAddress
			win1.textPhone.Value := cPhone
		End of Procedure
	Define Procedure firstrecord
		Code Block
			mymax := ADODB RecordSet (oRS) , Get Records Count
			IF ( mymax > 0 )
				Code Block
					ADODB RecordSet oRS - Goto First Record
					Call Procedure showrecord
				END OF IF STATEMENT
		End of Procedure
	Define Procedure lastrecord
		Code Block
			mymax := ADODB RecordSet (oRS) , Get Records Count
			IF ( mymax > 0 )
				Code Block
					ADODB RecordSet oRS - Goto Last Record
					Call Procedure showrecord
				END OF IF STATEMENT
		End of Procedure
	Define Procedure nextrecord
		Code Block
			mymax := ADODB RecordSet (oRS) , Get Records Count
			IF ( mymax > 0 )
				Code Block
					ADODB RecordSet oRS - Goto Next Record
					mycheck = ADODB RecordSet oRS , Check EOF
					IF ( mycheck )
						Code Block
							ADODB RecordSet oRS - Goto Previous Record
							Show Message "End of File" Title "Sorry" Type: Message Box
						ELSE
							Code Block
								Call Procedure showrecord
						END OF IF STATEMENT
				END OF IF STATEMENT
		End of Procedure
	Define Procedure prevrecord
		Code Block
			mymax := ADODB RecordSet (oRS) , Get Records Count
			IF ( mymax > 0 )
				Code Block
					ADODB RecordSet oRS - Goto Previous Record
					mycheck = ADODB RecordSet oRS , Check BOF
					IF ( mycheck )
						Code Block
							ADODB RecordSet oRS - Goto Next Record
							Show Message "Begin of File" Title "Sorry" Type: Message Box
						ELSE
							Code Block
								Call Procedure showrecord
						END OF IF STATEMENT
				END OF IF STATEMENT
		End of Procedure
