Procedures
	Define Procedure myadd
		Code Block
			mytext = win1.text1.Value
			mytext = CONVERT mytext TO Numeric
			win1.list1.AddItem ( mytext )
		End of Procedure
	Define Procedure mymin
		Code Block
			mycount = win1.list1.ItemCount
			IF ( mycount=0 )
				Code Block
					Show Message "Please, enter the numbers" Title "Sorry" Type: Message Box
					Return 0
				END OF IF STATEMENT
			mymin = win1.list1.item(1)
			IF ( mycount=1 )
				Code Block
					Show Message mymin Title "min number" Type: Message Box
					Return 0
				END OF IF STATEMENT
			mymin = CONVERT mymin TO Numeric
			FOR LOOP ( From x = 2 To mycount Step 1)
				Code Block
					myitem = win1.list1.item(x)
					myitem = CONVERT myitem TO Numeric
					IF ( myitem > mymin )
						Code Block
							mymin = myitem
						END OF IF STATEMENT
					mymin = CONVERT mymin TO STRING
				End of For Loop
		End of Procedure
